🧱 DamageComponent Model
class DamageComponent(models.Model):
    spell = FK Spell
    dice_count = int
    die_size = int
    damage_type = str
    timing = enum(
        "on_hit",
        "on_fail",
        "on_success",
        "end_of_turn",
        "per_round"
    )
    scales_with_slot = bool
